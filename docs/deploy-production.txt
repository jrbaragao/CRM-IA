# COMANDOS PARA PARAR O TESTE E SUBIR PRODUÇÃO

# 1. DELETAR O SERVIÇO DE TESTE:
gcloud run services delete analise-nf-teste --region us-central1 --quiet

# 2. FAZER DEPLOY DA APLICAÇÃO COMPLETA:
gcloud run deploy analise-nf --source . --platform managed --region us-central1 --allow-unauthenticated --memory 4Gi --cpu 4 --timeout 900 --min-instances 0 --max-instances 10

# 3. CONFIGURAR VARIÁVEL DE AMBIENTE (OPCIONAL):
# Se quiser configurar a API Key da OpenAI:
# gcloud run services update analise-nf --update-env-vars OPENAI_API_KEY=sk-sua-chave-aqui --region us-central1

# 4. VER A URL DO NOVO SERVIÇO:
# gcloud run services describe analise-nf --region us-central1 --format 'value(status.url)'

# 5. VER LOGS DO NOVO SERVIÇO:
# gcloud run services logs tail analise-nf --region us-central1

# COMANDOS ÚTEIS:

# gcloud run services list --region us-central1
